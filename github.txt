# File: .github/workflows/reusable_release.yml
name: Reusable Release Workflow

on:
  workflow_call:
    inputs:
      tag:
        description: 'Release tag'
        required: true
        type: string
    secrets:
      RELEASE_GITHUB_TOKEN:
        required: true

jobs:
  release:
    runs-on: ubuntu-latest
    steps:
      - name: Check Version Format
        id: version
        uses: nowsprinting/check-version-format-action@v4.0.4
        with:
          prefix: 'v'

      - name: Checkout repository
        if: steps.semver.outputs.valid == 'true'
        uses: actions/checkout@v4

      - name: Download Artifacts
        if: steps.semver.outputs.valid == 'true'
        uses: actions/download-artifact@v4
        with:
          path: artifacts/

      - name: Validate Artifacts
        if: steps.semver.outputs.valid == 'true'
        run: |
          if ! find artifacts/ -type f | read; then
            echo "❌ No artifacts found!"
            exit 1
          fi
          ls -lah artifacts/

      - name: Create GitHub Release
        if: steps.semver.outputs.valid == 'true'
        uses: softprops/action-gh-release@v2
        with:
          tag_name: ${{ inputs.tag }}
          title: "Release ${{ inputs.tag }}"
          notes_file: CHANGELOG.md
        env:
          GH_TOKEN: ${{ secrets.RELEASE_GITHUB_TOKEN }}

      - name: Upload Artifacts to Release
        if: steps.semver.outputs.valid == 'true'
        run: |
          find artifacts/ -type f -exec gh release upload "${{ inputs.tag }}" {} \;
        env:
          GH_TOKEN: ${{ secrets.RELEASE_GITHUB_TOKEN }}

File: workflows/build.yml
Content:
name: Build and Package

on:
  push:
    tags:
      - 'v*'
  pull_request:
  workflow_call:
    inputs:
      tag:
        description: 'Release tag'
        required: true
        type: string

env:
  NODE_VERSION: '20'

jobs:
  sync-version:
    runs-on: ubuntu-latest
    container:
      image: hakimjonas/fin-deb-build
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}

      - name: Install Rust Toolchain
        uses: actions-rust-lang/setup-rust-toolchain@v1
        with:
          toolchain: stable

      - name: Install cargo-edit
        run: cargo install cargo-edit

      - name: Check Version Format
        id: check_version
        uses: nowsprinting/check-version-format-action@v4.0.4
        with:
          prefix: 'v'

      - name: Sync Version
        if: steps.check_version.outputs.is_valid == 'true'
        uses: ./.github/actions/sync
        with:
          tag: ${{ github.ref_name }}

  build:
    needs: sync-version
    runs-on: ubuntu-latest
    container:
      image: hakimjonas/fin-deb-build
    strategy:
      fail-fast: false
      matrix:
        distro: [ arch, nix, solus, ubuntu ]
        include:
          - distro: arch
            package_task: arch
            artifact_path: target/fin-*-arch.tar.gz
          - distro: nix
            package_task: nix
            artifact_path: target/fin-*-nix.tar.gz
          - distro: solus
            package_task: solus
            artifact_path: target/fin-*-solus.tar.gz
          - distro: ubuntu
            package_task: deb
            artifact_path: target/debian/fin_*_amd64.deb
    env:
      GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
          fetch-tags: true

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}

      - name: Install Rust Toolchain
        uses: actions-rust-lang/setup-rust-toolchain@v1
        with:
          toolchain: stable

      - name: Install cargo-make
        run: cargo install cargo-make

      - name: Cache Cargo dependencies
        uses: actions/cache@v4
        with:
          path: |
            ~/.cargo/registry
            ~/.cargo/git
            target
          key: ${{ runner.os }}-cargo-${{ hashFiles('**/Cargo.lock') }}
          restore-keys: |
            ${{ runner.os }}-cargo-

      - name: Build Release using Cargo Make
        run: cargo make build-release

      - name: Package using Cargo Make
        run: cargo make package-${{ matrix.package_task }}

      - name: Verify artifacts
        run: |
          if [ "${{ matrix.distro }}" = "ubuntu" ]; then
            PATTERN="./target/debian/fin_*_amd64.deb"
          else
            PATTERN="./${{ matrix.artifact_path }}"
          fi
          echo "Verifying artifact using find with pattern: $PATTERN"
          if ! find . -type f -path "$PATTERN" | grep -q .; then
            echo "Artifact not found"
            exit 1
          fi

      - name: Upload artifacts
        if: ${{ !env.ACT }}  # Skip upload when running locally with act
        uses: actions/upload-artifact@v4
        with:
          name: fin-${{ matrix.distro }}-${{ github.ref_name }}
          path: ${{ matrix.artifact_path }}\n--------------------------------\n
